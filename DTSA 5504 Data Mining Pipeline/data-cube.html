def plot_data_cube():
    '''
        Output: x, y, z
    '''
    # Reading dataset
    df = pd.read_csv("./dataset/dataset.csv")
    
    # your code here
    d = df.groupby(["year","category"])["sales"].sum().reset_index()
    x = list( d.loc[:,"year"] )
    y = list( d.loc[:,"category"] )
    z = list( d.loc[:,"sales"] )
    
    return x,y,z
